import error from './error' import {countif} from './countif' export function countif(...values) {

values must be provided in pairs

if (values.length % 2 !== 0) { return error.na } return values.reduce( (acc, item, index) => {

skip odd rows

if (index % 2 !== 0) { return acc; } var range = item, criteria = values[index+1] sum = countif(range, criteria) return acc + sum; }, 0) }
